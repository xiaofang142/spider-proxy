```dart
import 'package:purchases_flutter/purchases_flutter.dart';

class SubscriptionManager {
  static const String API_KEY = 'your_revenuecat_api_key';

  Future<void> initialize() async {
    await Purchases.setLogLevel(LogLevel.debug);
    await Purchases.configure(API_KEY);

        offerings.current!.availablePackages[0]
      );
    }
  }

  Future<void> restorePurchases() async {
    await Purchases.restorePurchases();
  }
}
```

### 4.2 Platform Channel 开发

**场景**: Flutter 调用 Android 原生功能

**步骤 1: Android 端实现方法**

```kotlin
// MainActivity.kt
class MainActivity : FlutterActivity() {
    override fun configureFlutterEngine(flutterEngine: FlutterEngine) {
        super.configureFlutterEngine(flutterEngine)

        MethodChannel(flutterEngine.dartExecutor.binaryMessenger, "spider_proxy/utils")
            .setMethodCallHandler { call, result ->
                when (call.method) {
                    "getDeviceInfo" -> handleGetDeviceInfo(result)
                    "checkPermission" -> handleCheckPermission(call, result)
                    else -> result.notImplemented()
                }
            }
    }

    private fun handleGetDeviceInfo(result: MethodChannel.Result) {
        val deviceInfo = mapOf(
            "model" to Build.MODEL,
            "version" to Build.VERSION.RELEASE,
            "sdk" to Build.VERSION.SDK_INT
        )
        result.success(deviceInfo)
    }
}
```

**步骤 2: Flutter 端调用**

```dart
// lib/utils/device_info.dart
class DeviceInfo {
  static const MethodChannel _channel = MethodChannel('spider_proxy/utils');

  static Future<Map<String, dynamic>> getInfo() async {
    try {
      final result = await _channel.invokeMethod<Map>('getDeviceInfo');
      return result ?? {};
    } on PlatformException catch (e) {
      throw Exception('获取设备信息失败：${e.message}');
    }
  }
}

// 使用
final info = await DeviceInfo.getInfo();
print('设备型号：${info['model']}');
```

## 七、常见问题

### 7.1 VPN 服务启动失败

**问题**: `SecurityException: Permission denied`

**解决**:
```xml
<!-- 确保 AndroidManifest.xml 包含以下权限 -->
<uses-permission android:name="android.permission.BIND_VPN_SERVICE" />
<uses-permission android:name="android.permission.FOREGROUND_SERVICE" />
<uses-permission android:name="android.permission.POST_NOTIFICATIONS" />
```

### 7.2 证书安装后不信任

**问题**: HTTPS 解密失败，显示证书错误

**解决**:
1. 确保证书已安装到**系统信任存储** (而非用户证书)
2. Android 7.0+ 需要 root 权限才能安装系统证书
3. 或者使用 Virtual App 方案 (如 VirtualXposed)

### 7.3 端口被占用

**问题**: `Address already in use`

**解决**:
```dart
// 检查端口占用
bool isPortAvailable(int port) async {
  try {
    final server = await ServerSocket.bind('127.0.0.1', port);
    await server.close();
    return true;
  } catch (e) {
    return false;
  }
}

// 使用时
final port = 8888;
if (!await isPortAvailable(port)) {
  // 提示用户或自动选择下一个可用端口
  showPortInUseDialog();
}
```

### 7.4 内存占用过高

**问题**: 应用运行一段时间后内存占用超过 200MB

**解决**:
```kotlin
// 1. 实现动态缓冲区
class DynamicBuffer {
  var bufferSize = 16 * 1024
  val maxBufferSize = 1 * 1024 * 1024

  fun adjust(isOverflow: Boolean) {
    if (isOverflow && bufferSize < maxBufferSize) {
      bufferSize = minOf(bufferSize * 2, maxBufferSize)
    }
  }
}

// 2. 定期清理旧记录
class TrafficRepository {
  Future<void> cleanupOldRecords() async {
    final sevenDaysAgo = DateTime.now().subtract(Duration(days: 7));
    await database.delete(
      'traffic_records',
      'timestamp < ?',
      [sevenDaysAgo.millisecondsSinceEpoch],
    );
  }
}
```
