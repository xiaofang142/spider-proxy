```kotlin
// Android KeyStore 存储私钥
class SecureKeyStore(private val context: Context) {
    private val keyStore = KeyStore.getInstance("AndroidKeyStore")

    fun initialize() {
        keyStore.load(null)
        if (!keyStore.containsAlias("spider_proxy_ca")) {
            generateKey()
        }
    }

    private fun generateKey() {
        val keyGenerator = KeyGenerator.getInstance(
            KeyProperties.KEY_ALGORITHM_RSA,
            "AndroidKeyStore"
        )
        keyGenerator.init(
            KeyGenParameterSpec.Builder(
                "spider_proxy_ca",
                KeyProperties.PURPOSE_SIGN
            )
                .setKeySize(2048)
                .setDigests(KeyProperties.DIGEST_SHA256)
                .setEncryptionPaddings("PKCS1v15")
                .build()
        )
        keyGenerator.generateKeyPair()
    }

    fun getPrivateKey(): PrivateKey {
        val entry = keyStore.getEntry("spider_proxy_ca", null) as KeyStore.PrivateKeyEntry
        return entry.privateKey
    }
}
```

```dart
// 完整 MITM 解密实现
class MitmProxy {
  // 处理 HTTPS CONNECT 请求
  Future<void> _handleHttpsConnect(HttpRequest request) async {
    final host = request.uri.host;
    final port = request.uri.port;

    // 1. 获取主机 SSL 上下文（动态生成证书）
    final sslContext = await _certManager.getSSLContextForHost(host);

    // 2. 发送 200 响应给客户端
    await request.response
        ..statusCode = HttpStatus.ok
        ..reasonPhrase = 'Connection Established'
        ..close();

    // 3. 与客户端建立 SSL 连接
    final clientSocket = request.connectionInfo!.socket;
    final secureSocket = await sslContext.startServerSocketConnection(
      clientSocket,
      requestClientAllowed: true,
    );

    // 4. 与目标服务器建立 SSL 连接
    final targetSocket = await SecureSocket.connect(host, port);

    // 5. 双向解密转发
    await _decryptAndForward(secureSocket, targetSocket, request);
  }

  // 双向解密转发
  Future<void> _decryptAndForward(
    SecureSocket client,
    SecureSocket target,
    HttpRequest request,
  ) async {
    final completer = Completer<void>();

    // 客户端 -> 目标服务器
    client.listen((data) {
      target.add(data);
      // 记录请求数据
      _recordRequest(request, data);
    }, onDone: () {
      target.close();
    });

    // 目标服务器 -> 客户端
    target.listen((data) {
      client.add(data);
      // 记录响应数据
      _recordResponse(request, data);
    }, onDone: () {
      client.close();
      completer.complete();
    });

    return completer.future;
  }
}
```

```dart
class JsonViewer extends StatelessWidget {
  final String jsonString;

  @override
  Widget build(BuildContext context) {
    try {
      final jsonObj = json.decode(jsonString);
      return SingleChildScrollView(
        child: JsonTreeView(data: jsonObj),
      );
    } catch (e) {
      return Text(jsonString); // 非 JSON 原样显示
    }
  }
}

class JsonTreeView extends StatelessWidget {
  final dynamic data;
  final bool expanded;

  @override
  Widget build(BuildContext context) {
    if (data is Map) {
      return Column(
        children: data.entries.map((e) => JsonEntry(key: e.key, value: e.value)).toList(),
      );
    }
    return Text(JsonEncoder.withIndent('  ').convert(data));
  }
}
```

```dart
class SubscriptionManager {

  Future<void> initialize() async {
    await Purchases.setLogLevel(LogLevel.debug);
    await Purchases.configure(API_KEY);

  Future<bool> isPremium() async {
    final customerInfo = await Purchases.getCustomerInfo();
    return customerInfo.entitlements.active.containsKey('premium');

  Future<void> purchasePremium() async {
    final offerings = await Purchases.getOfferings();
    if (offerings.current != null) {
      await Purchases.purchasePackage(offerings.current!.availablePackages[0]);
    }
  }

  Future<void> restorePurchases() async {
    await Purchases.restorePurchases();
  }
}
```
