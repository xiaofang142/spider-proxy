**版本**: 2.0

**架构类型**: Flutter + Android 原生混合架构

## 一、架构概述

### 1.1 架构目标

- **跨平台**: Flutter UI + Android/iOS 原生能力
- **高性能**: 非阻塞 I/O，协程并发，低延迟抓包
- **安全性**: KeyStore 私钥存储，证书自动过期
- **可扩展**: 清晰分层，模块解耦，便于功能迭代

### 1.2 技术栈

| 层级 | 技术选型 | 说明 |
|------|---------|------|
| UI 框架 | Flutter 3.24+ | 跨平台 UI，支持 Android + iOS |
| 状态管理 | flutter_redux | 全局状态管理 |
| 原生通信 | Platform Channel | Flutter ↔ 原生双向通信 |
| Android | Kotlin + Coroutines | 原生 VPN 服务、证书管理 |
| iOS (v2.0) | Swift + NetworkExtension | Packet Tunnel Provider |
| 加密库 | Bouncy Castle 1.70 | X.509 证书生成 |
| 非阻塞 I/O | Java NIO Selector | 高并发连接处理 |
| 数据存储 | SQLite (sqflite) | 本地流量记录存储 |
| 内购 | purchases_flutter | RevenueCat 订阅管理 |

```kotlin
// SslInterceptor.kt - HTTPS 解密
class SslInterceptor(private val certsDir: File) {
    // CA 证书
    private lateinit var caKeyPair: KeyPair
    private lateinit var caCertificate: X509Certificate

    // 证书缓存 (24 小时过期)
    private val certCache = ConcurrentHashMap<String, CachedCert>()

    // 初始化
    fun initialize() {
        // 从 KeyStore 加载或生成 CA 密钥对
        val keyStore = KeyStore.getInstance("AndroidKeyStore")
        keyStore.load(null)

        if (!keyStore.containsAlias("spider_proxy_ca")) {
            generateCaKeyPair(keyStore)
        } else {
            loadCaKeyPair(keyStore)
        }

        // 加载 CA 证书
        loadCaCertificate()
    }

    // 为主机生成证书
    fun getSSLContextForHost(host: String): SSLContext? {
        // 检查缓存
        val cached = certCache[host]
        if (cached != null && !cached.isExpired()) {
            return cached.sslContext
        }

        // 生成主机证书
        val hostCert = generateHostCertificate(host)
        val sslContext = createSSLContext(hostCert)

        // 缓存 (24 小时后过期)
        certCache[host] = CachedCert(
            sslContext = sslContext,
            expires = System.currentTimeMillis() + 24 * 60 * 60 * 1000
        )

        return sslContext
    }
}
```

### 5.1 CA 私钥安全存储

```kotlin
// 使用 Android KeyStore 存储私钥
class SecureKeyStore(private val context: Context) {
    private val keyStore = KeyStore.getInstance("AndroidKeyStore")

    fun initialize() {
        keyStore.load(null)

        if (!keyStore.containsAlias("spider_proxy_ca")) {
            generateKey()
        }
    }

    private fun generateKey() {
        val keyGenerator = KeyGenerator.getInstance(
            KeyProperties.KEY_ALGORITHM_RSA,
            "AndroidKeyStore"
        )

        keyGenerator.init(
            KeyGenParameterSpec.Builder(
                "spider_proxy_ca",
                KeyProperties.PURPOSE_SIGN
            )
                .setKeySize(2048)
                .setDigests(KeyProperties.DIGEST_SHA256)
                .setEncryptionPaddings("PKCS1v15")
                .setUserAuthenticationRequired(false)
                .build()
        )

        keyGenerator.generateKeyPair()
    }

    fun getPrivateKey(): PrivateKey {
        val entry = keyStore.getEntry("spider_proxy_ca", null)
            as KeyStore.PrivateKeyEntry
        return entry.privateKey
    }
}
```

```kotlin
// 证书 24 小时自动过期
data class CachedCert(
    val sslContext: SSLContext,
    val expires: Long  // 过期时间戳
) {
    fun isExpired(): Boolean = System.currentTimeMillis() > expires
}

// 在 SslInterceptor 中
fun getSSLContextForHost(host: String): SSLContext? {
    val cached = certCache[host]
    if (cached != null) {
        if (!cached.isExpired()) {
            return cached.sslContext  // 使用缓存
        } else {
            certCache.remove(host)    // 过期清除
        }
    }

    // 生成新证书 (24 小时后过期)
    val sslContext = generateNewSSLContext(host)
    certCache[host] = CachedCert(
        sslContext = sslContext,
        expires = System.currentTimeMillis() + 24 * 60 * 60 * 1000
    )

    return sslContext
}
```

## 六、性能设计

### 6.1 非阻塞 I/O

```kotlin
// 使用 Java NIO Selector 处理并发连接
class TcpForwarder {
    private val selector = Selector.open()

    init {
        // 后台线程监听连接
        vpnScope.launch {
            while (isActive) {
                val readyChannels = selector.select(1000)
                if (readyChannels > 0) {
                    val selectedKeys = selector.selectedKeys().iterator()
                    while (selectedKeys.hasNext()) {
                        val key = selectedKeys.next()
                        selectedKeys.remove()

                        if (key.isReadable) {
                            readFromChannel(key.channel() as SocketChannel)
                        }
                        if (key.isWritable) {
                            writeToChannel(key.channel() as SocketChannel)
                        }
                    }
                }
            }
        }
    }
}
```

### 6.2 连接池优化

```kotlin
// 连接复用，减少延迟
class ConnectionPool {
    private val pool = ConcurrentHashMap<String, Queue<SocketChannel>>()
    private val maxPoolSize = 50

    fun acquireConnection(host: String, port: Int): SocketChannel {
        val key = "$host:$port"
        val queue = pool[key]

        return if (!queue.isNullOrEmpty()) {
            queue.poll()  // 复用连接
        } else {
            createNewConnection(host, port)
        }
    }

    fun returnConnection(host: String, port: Int, channel: SocketChannel) {
        val key = "$host:$port"
        val queue = pool.computeIfAbsent(key) { ConcurrentLinkedQueue() }

        if (queue.size < maxPoolSize) {
            queue.offer(channel)  // 归还连接
        } else {
            channel.close()       // 池满关闭
        }
    }
}
```

### 6.3 动态缓冲区

```kotlin
// 根据流量大小自动调整缓冲区
class DynamicBuffer {
    private var bufferSize = 16 * 1024  // 初始 16KB
    private val maxBufferSize = 1 * 1024 * 1024  // 最大 1MB
    private var overflowCount = 0

    fun adjust(currentSize: Int, isOverflow: Boolean) {
        if (isOverflow) {
            overflowCount++

            // 连续溢出，扩大缓冲区
            if (overflowCount >= 3 && bufferSize < maxBufferSize) {
                bufferSize = minOf(bufferSize * 2, maxBufferSize)
                overflowCount = 0
            }
        } else {
            // 正常情况，重置计数
            overflowCount = 0
        }
    }

    fun getBuffer(): ByteBuffer = ByteBuffer.allocate(bufferSize)
}
```

```swift
    func getPrivateKey() -> SecKey? {

            kSecClass as String: kSecClassKey,
            kSecAttrApplicationTag as String: "spider_proxy_ca".data(using: .utf8)!,

            return item as? SecKey
        }

            kSecClass as String: kSecClassKey,

            kSecValueRef as String: key

        return SecItemAdd(query as CFDictionary, nil)
```

```dart
    return vpnEventChannel.receiveBroadcastStream().map((event) {
      return ProxyStatus.values.firstWhere(
        (s) => s.index == event['status'],
        orElse: () => ProxyStatus.stopped,
      );
    });
  }
}
```
