## 一、架构概述

### 1.1 设计原则

1. **轻量级**: 核心功能精简，避免过度设计
2. **模块化**: 高内聚低耦合，便于扩展维护
3. **性能优先**: 抓包延迟 < 100ms，内存 < 200MB
4. **用户友好**: 配置简单，操作直观
5. **开源友好**: 代码清晰，文档完善

```kotlin
class SpiderVpnService : VpnService() {
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        val builder = VpnService.Builder()
            .addAddress("10.0.0.1", 32)
            .addRoute("0.0.0.0", 0)
            .addDnsServer("8.8.8.8")
            .setSession("Spider Proxy")
        
        val vpnInterface = builder.establish()
        // 转发流量到代理服务器
        forwardToProxy(vpnInterface, "127.0.0.1", 8888)
        
        return START_STICKY
    }
}

// MITM 代理服务器
class MitmProxyServer(
    private val port: Int,
    private val certManager: CertManager
) {
    private val server = NettyHttpServer()
    
    fun start() {
        server.bind(port) { request ->
            handleRequest(request)
        }
    }
    
    private suspend fun handleRequest(request: HttpRequest): HttpResponse {
        // SSL 解密
        if (request.isHttps) {
            val sslContext = certManager.getSslContext(request.host)
            return decryptAndForward(request, sslContext)
        }
        
        // 普通 HTTP 转发
        return forward(request)
    }
}

// 流量处理器
class FlowProcessor {
    private val flowList = mutableListOf<Flow>()
    private val listeners = mutableListOf<FlowListener>()
    
    fun process(rawData: ByteArray): Flow {
        val flow = parseFlow(rawData)
        flowList.add(flow)
        notifyListeners(flow)
        return flow
    }
    
    fun addListener(listener: FlowListener) {
        listeners.add(listener)
    }
}
```

```kotlin
class CertManager(private val context: Context) {
    private val caCertPath = File(context.filesDir, "ca.crt")
    private val caKeyPath = File(context.filesDir, "ca.key")
    private val certCache = mutableMapOf<String, X509Certificate>()
    
    // 初始化 CA 证书
    fun initCaCert() {
        if (!caCertPath.exists()) {
            generateCaCert()
        }
    }
    
    // 生成 CA 证书
    private fun generateCaCert() {
        val keyPair = generateKeyPair()
        val cert = generateSelfSignedCert(keyPair, "Spider Proxy CA")
        
        caKeyPath.writeText(keyPair.private.encodeToString())
        caCertPath.writeText(cert.encodeToString())
    }
    
    // 获取站点 SSL 上下文
    fun getSslContext(host: String): SSLContext {
        return certCache.getOrPut(host) {
            generateSiteCert(host)
        }.let { cert ->
            createSslContext(cert)
        }
    }
    
    // 安装 CA 证书
    fun installCaCert(): Boolean {
        val intent = Intent(CredentialInstallerAction)
        intent.putExtra(CredentialInstallerExtra, caCertPath.readBytes())
        intent.setFlags(FLAG_ACTIVITY_NEW_TASK)
        context.startActivity(intent)
        return true
    }
}
```

```kotlin
@Composable
fun RequestDetail(flow: Flow) {
    Column(modifier = Modifier.verticalScroll(rememberScrollState())) {
        // 基本信息
        InfoRow(label = "URL", value = flow.url)
        InfoRow(label = "Method", value = flow.method)
        InfoRow(label = "Status", value = flow.responseStatus.toString())
        InfoRow(label = "Time", value = formatTime(flow.timestamp))
        InfoRow(label = "Size", value = formatSize(flow.size))
        InfoRow(label = "Duration", value = "${flow.duration}ms")
        
        // 请求头
        SectionTitle(title = "Headers")
        flow.requestHeaders.forEach { (key, value) ->
            HeaderRow(key = key, value = value)
        }
        
        // 请求体
        if (flow.requestBody != null) {
            SectionTitle(title = "Body")
            BodyViewer(
                data = flow.requestBody,
                contentType = flow.requestHeaders["Content-Type"]
            )
        }
    }
}

@Composable
fun BodyViewer(
    data: ByteArray,
    contentType: String?
) {
    val formattedText = remember(data, contentType) {
        formatBody(data, contentType)
    }
    
    when {
        contentType?.contains("image") == true -> {
            AsyncImage(
                model = ImageRequest.Builder(LocalContext.current)
                    .data(data)
                    .build(),
                contentDescription = "Response Image"
            )
        }
        contentType?.contains("json") == true -> {
            CodeText(
                text = formattedText,
                language = "json"
            )
        }
        else -> {
            Text(text = formattedText)
        }
    }
}
```

```kotlin
// 流量列表分页加载
class FlowRepository {
    private val flowCache = ConcurrentLinkedDeque<Flow>()
    private val maxCacheSize = 1000
    
    fun addFlow(flow: Flow) {
        flowCache.addFirst(flow)
        
        // 超过阈值，移除旧流量
        while (flowCache.size > maxCacheSize) {
            flowCache.removeLast()
        }
    }
    
    fun getFlows(page: Int, pageSize: Int): List<Flow> {
        return flowCache
            .drop(page * pageSize)
            .take(pageSize)
    }
}
```
